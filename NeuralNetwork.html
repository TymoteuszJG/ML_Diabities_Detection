class SimpleNN(nn.Module):
    def __init__(self, input_size, hidden_size, output_size, dropout_rate=0.5):
        super(SimpleNN, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size * 2)
        self.relu1 = nn.ReLU()
        self.dropout1 = nn.Dropout(p=dropout_rate)
        self.fc2 = nn.Linear(hidden_size * 2, hidden_size)
        self.relu2 = nn.ReLU()
        self.dropout2 = nn.Dropout(p=dropout_rate)
        self.fc3 = nn.Linear(hidden_size, output_size)
        self.sigmoid = nn.Sigmoid()

    def forward(self, x):
        x = self.fc1(x)
        x = self.relu1(x)
        x = self.dropout1(x)
        x = self.fc2(x)
        x = self.relu2(x)
        x = self.dropout2(x)
        x = self.fc3(x)
        x = self.sigmoid(x)
        return x

In [9]:
X_tensor = torch.tensor(X_scaled, dtype=torch.float32)
y_tensor = torch.tensor(y.values, dtype=torch.float32).view(-1, 1) 
X_train, X_test, y_train, y_test = train_test_split(X_tensor, y_tensor, test_size=0.3, random_state=42)

positive_class_weight = torch.tensor([(y_train == 0).sum().item() / (y_train == 1).sum().item()], dtype=torch.float32)

criterion = nn.BCEWithLogitsLoss(pos_weight=positive_class_weight)

learning_rates = [0.0001, 0.001, 0.01, 0.1]

accuracy_scores = []
precision_scores = []
recall_scores = []
f1_scores = []

for lr in learning_rates:
    model = SimpleNN(input_size=21, hidden_size=20, output_size=1)
    optimizer = optim.Adam(model.parameters(), lr=lr)
    
    num_epochs = 100
    for epoch in range(num_epochs):
        outputs = model(X_train)
        y_train_reshaped = y_train.view(-1)
        loss = criterion(outputs.view(-1), y_train_reshaped)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

    with torch.no_grad():
        model.eval() 
        y_pred = model(X_test)
        y_pred_class = torch.round(y_pred)

    y_test_np = y_test.numpy()
    y_pred_class_np = y_pred_class.numpy()

    accuracy = accuracy_score(y_test_np, y_pred_class_np)
    precision = precision_score(y_test_np, y_pred_class_np)
    recall = recall_score(y_test_np, y_pred_class_np)
    f1 = f1_score(y_test_np, y_pred_class_np)

    accuracy_scores.append(accuracy)
    precision_scores.append(precision)
    recall_scores.append(recall)
    f1_scores.append(f1)
