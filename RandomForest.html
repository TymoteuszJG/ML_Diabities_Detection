data=pd.read_csv('diabetes_binary_health_indicators_BRFSS2015.csv')
X = data.drop("Diabetes_binary", axis=1)
y = data["Diabetes_binary"] 
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)
n_estimators_list=np.arange(10, 150,10)

accuracy_scores_rf = []
recall_scores_rf = []
precision_scores_rf = []
f1_scores_rf = []

for n_estimators in n_estimators_list:
    rf_model = RandomForestClassifier(n_estimators=n_estimators,max_depth=20, random_state=42)
    rf_model.fit(X_train, y_train)
    rf_predictions = rf_model.predict(X_test)

    accuracy_scores_rf.append(accuracy_score(y_test, rf_predictions))
    recall_scores_rf.append(recall_score(y_test, rf_predictions))
    precision_scores_rf.append(precision_score(y_test, rf_predictions, zero_division=0))
    f1_scores_rf.append(f1_score(y_test, rf_predictions))

plt.figure(figsize=(12, 8))

plt.subplot(2, 2, 1)
plt.plot(n_estimators_list, accuracy_scores_rf)
plt.title('Accuracy (Random Forest)')
plt.xlabel('Number of Trees')
plt.ylabel('Accuracy')

plt.subplot(2, 2, 2)
plt.plot(n_estimators_list, recall_scores_rf)
plt.title('Recall (Random Forest)')
plt.xlabel('Number of Trees')
plt.ylabel('Recall')

plt.subplot(2, 2, 3)
plt.plot(n_estimators_list, precision_scores_rf)
plt.title('Precision (Random Forest)')
plt.xlabel('Number of Trees')
plt.ylabel('Precision')

plt.subplot(2, 2, 4)
plt.plot(n_estimators_list, f1_scores_rf)
plt.title('F1 Score (Random Forest)')
plt.xlabel('Number of Trees')
plt.ylabel('F1 Score')

plt.tight_layout()
plt.show()
